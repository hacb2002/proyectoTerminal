int potencia(int a, int b){
    if(b == 0){
        return 1;
    }else{
        int t = potencia(a, b/2);
        if(b % 2 == 0){
            return t*t;
        }else{
            return a*t*t;
        }
    }
}

int f(int n){
  if(n <= 1){
    return 0;
  }else{
    return 2*f(n-1)+potencia(2,n-1);
  }
}
