int potencia(int a, int b){
    if(b == 0){
        return 1;
    }else{
        int t = potencia(a, b/2);
        if(b % 2 == 0){
            return t*t;
        }else{
            return a*t*t;
        }
    }
}

int conteo(int n, int k){
   if(k == 0){
      return potencia(9,n);
   }
   else if(n == 0){
    return 1;
   }
   else if(n == 1){ 
    return 10;
   }
   else return 9*(conteo(n-1,k)+conteo(n-2,k-1));
}
