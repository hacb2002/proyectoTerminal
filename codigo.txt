/* Función de ejemplo en C que demuestra comentarios,
   declaraciones de variables y varias comparaciones */
int procesarValores(int valor1, int valor2, char modo) {
    
    // Declaración de variables principales
    int resultado    = 0;
    int suma         = 0;
    int diferencia   = 0;
    char estado      = modo;  // modo de operación
    
    /* Calculamos suma y diferencia */
    suma       = valor1 + valor2;
    diferencia = valor1 - valor2;
    
    // Primera comparación: ¿qué número es mayor?
    if (valor1 > valor2) {
        resultado = valor1;
    }
    else if (valor1 < valor2) {
        resultado = valor2;
    }
    else {
        // Si son iguales, usamos la suma
        resultado = suma;
    }
    
    /* Ajustes según el 'modo' pasado:
       - modo 'A': añadimos 10
       - modo 'B': restamos 5 si no son iguales
       - cualquier otro: devolvemos diferencia absoluta */
    if ( estado == 'A' || estado == 'a' ) {
        resultado = 10+5;
    }
    else if ( estado == 'B' && valor1 != valor2 ) {
        resultado = 5;
    }
    else {
        if ( diferencia < 0 ) {
            diferencia = -diferencia;  // valor absoluto
        }
        resultado = diferencia;
    }
    
    // Devolvemos el resultado final
    return resultado;
}
